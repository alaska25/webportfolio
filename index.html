import asyncio, torch
from .cluster import NodePool
 
class TrainingScheduler:
  def __init__(self, pool: NodePool):
    self.pool = pool
    self.queue = asyncio.Queue()
 
  async def dispatch(self, job):
    node = await self.pool.allocate(
      gpus=job.gpu_count,
      memory=32768
    )
    return await node.run(job)
